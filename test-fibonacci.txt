# SECUENCIA DE FIBONACCI
# ===================================
# Calcula los primeros 10 números de Fibonacci
# F(0)=0, F(1)=1, F(n)=F(n-1)+F(n-2)
# Guarda cada valor en memoria
# x5 = F(n-2) anterior
# x6 = F(n-1) actual  
# x7 = F(n) nuevo
# x8 = contador
# x9 = límite (10 números)
# x10 = índice memoria
# ===================================

addi x5, x0, 0
addi x6, x0, 1
addi x8, x0, 0
addi x9, x0, 10
addi x10, x0, 0
sw x5, 0(x10)
addi x10, x10, 4
addi x8, x8, 1
beq x8, x9, 36
add x7, x5, x6
sw x7, 0(x10)
addi x10, x10, 4
add x5, x0, x6
add x6, x0, x7
addi x8, x8, 1
beq x8, x9, 8
beq x0, x0, -28
beq x0, x0, 0
